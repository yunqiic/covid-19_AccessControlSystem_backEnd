### 一、后台登录

#### ① 后台登录验证

```java
url: '/admin/login'
method: 'post'
params: {
    username: '',		//用户名
    password: ''		//密码
}
response: {
    //验证成功
    code: 20000
    data: 'admin-token'		//规定字符串
    //验证失败
    message: '用户名 或 密码错误.'
}
```



#### ② 后台登录token，默认需要的接口，返回如下规定的json字符串即可

```
url: '/admin/info'
method: 'get'
response: {
    code: 20000
    data: {		//已规定好
        roles: ['admin'],
        introduction: 'I am a super administrator',
        avatar: 'https://wpimg.wallstcn.com/f778738c-e4f8-4870-b634-56703b4acafe.gif',
        name: 'Super Admin'
    }
}
```



### **二、账号管理**



#### **1.  管理员(admin)**

##### ① 按筛选条件获取管理员列表

```java
url: '/account/admin/list'
method: 'get'
params: {
    id: '',             //根据id查询，为空时不过滤，字符串!!! 
    username: '',       //根据用户名查询，为空时不过滤
    page: 1,            //页数
    limit: 20,          //每页个数
    sort: '+id'         //排序，为'-id'时逆序
}
response: {
    code: 20000,
    data: {
        list: {         //返回对象数组，根据limit确认数组大小，默认每页20个，list[20]
        	{ id: 123, username: 'test1' },
        	{ id: 456, username: 'test2' }
    	}
    	total: 200      //符合条件的总记录数
	}
}
```

##### ② 添加管理员

```java
url: '/account/admin/create'
method: 'post'
params: {
    username: 'test',           //用户名
    password: 'test',           //密码
    comfirm_password: 'test'    //确认密码
}
response: {
    code: 20000,
    data: {
        id: int		//新添加管理员的id
    	result: 0   //信号量  成功:0   用户名已存在:1   两次密码不一致:2   读写错误:3
    }
}
```

##### ③ 修改管理员密码

```java
url: '/account/admin/update'
method: 'post'
params: {
    id: '123',                  //管理员id
    old_password: 'test'		//旧密码
    new_password: 'test',       //新密码
    comfirm_password: 'test'    //确认密码
}
response: {
    code: 20000,
    data: {
    	result: 0	//信号量  成功:0   两次密码不一致:1   旧密码不正确:2   读写错误:3
    }
}
```

##### ④ 删除管理员

```java
url: '/account/admin/delete'
method: 'post'
params: {
    id: '123'   //管理员id
}
response: {
    code: 20000,
    data: {
    	result: 0	//信号量  成功:0   失败:1
    }
}
```

### 三、出入记录



#### ① 按筛选条件获取出入记录列表

```java
url: '/record/list'
method: 'get'
params: {
    id: '',             //根据记录id查询，为空时不过滤，字符串!!! 
    resident_id: '',    //根据住户id查询，为空时不过滤，字符串!!! 
    inspector_id: '',   //根据检查员id查询，为空时不过滤，字符串!!!
    date_query: '',     /*
                        根据日期查询，为空时不过滤
                        日期时间戳数组date_query[2]
                        date_query[0]为开始日期，date_query[1]为结束日期
                        */
    health_query: '',   /*
                        根据体温查询，为空时不过滤，字符串!!! 
                        为'healthy' 过滤 temperature < 发烧温度
                        为'unhealthy' 过滤 temperature >= 发烧温度
                        */
    page: 1,            //页数
    limit: 20,          //每页个数
    sort: '+id'         //排序，为'-id'时逆序
}
response: {
    code: 20000,
    data: {
        list: {         //返回对象数组，根据limit确认数组大小，默认每页20个，list[20]
        { 
            id: 123,                //记录id
            time: 90997278243,      //日期时间戳
            temperature: 37.5,      //体温
            resident_id: 123,       //住户id
            inspector_id: 123       //检查员id
        },
        {
            ...
        }
    }
}
```

### 四、统计模块



#### 1. 健康统计

##### ① 统计 健康 与 高危 的住户人数

逻辑概要：

连续高温天数 health_status >= 预警值 为高危

连续高温天数 health_status < 预警值 为健康

```java
url: '/analysis/health'
method: 'get'
response: {
    code: 20000,
    data: {
        health_count: [50,100]	//index[0]健康住户人数,index[1]高危住户人数
    }
}
```



#### 2. 出入统计

##### ① 统计 最近 7或30 天 每天 出入记录条数

```java
url: '/analysis/record'
method: 'get'
params: {
    query:{
        option: '7'或'30'	// 为7时统计最近7天，为30时统计最近30天，字符串！！！
    }
}
response: {
    code: 20000,
    data: {
		access_dates: ['2018-06-01', ... , '2018-06-07'],	// 最近7或30天日期
		access_times: [1, ... , 7]	// 7或30天分别对应的出入记录条数
    }
}
```
